`timescale 1ns/1ps

module traffic_light_controller
#(
    parameter GREEN_TIME  = 5,
    parameter YELLOW_TIME = 2
)
(
    input clk,
    input reset,
    input emergency,

    output reg [1:0] NS,
    output reg [1:0] EW
);

// Light Encoding
parameter RED    = 2'b00;
parameter YELLOW = 2'b01;
parameter GREEN  = 2'b10;

// FSM States
parameter NS_GREEN  = 2'd0;
parameter NS_YELLOW = 2'd1;
parameter EW_GREEN  = 2'd2;
parameter EW_YELLOW = 2'd3;

reg [1:0] current_state;
reg [1:0] next_state;

reg [3:0] timer;

// State Register

always @(posedge clk or posedge reset)
begin

    if(reset)
        current_state <= NS_GREEN;

    else if(emergency)
        current_state <= NS_GREEN;

    else
        current_state <= next_state;

end

// Timer

always @(posedge clk or posedge reset)
begin

    if(reset)

        timer <= 0;

    else if(emergency)

        timer <= 0;

    else
    begin

        if(current_state != next_state)
            timer <= 0;
        else
            timer <= timer + 1;

    end

end


// Next State Logic

always @(*)
begin

    next_state = current_state;

    case(current_state)

    NS_GREEN:
    begin

        if(timer >= GREEN_TIME)
            next_state = NS_YELLOW;

    end

    NS_YELLOW:
    begin

        if(timer >= YELLOW_TIME)
            next_state = EW_GREEN;

    end

    EW_GREEN:
    begin

        if(timer >= GREEN_TIME)
            next_state = EW_YELLOW;

    end

    EW_YELLOW:
    begin

        if(timer >= YELLOW_TIME)
            next_state = NS_GREEN;

    end

    default:

        next_state = NS_GREEN;

    endcase

end
  
// Output Logic (Moore FSM)

always @(*)
begin

    NS = RED;
    EW = RED;

    if(emergency)
    begin

        NS = GREEN;
        EW = RED;

    end

    else
    begin

        case(current_state)

        NS_GREEN:
        begin
            NS = GREEN;
            EW = RED;
        end

        NS_YELLOW:
        begin
            NS = YELLOW;
            EW = RED;
        end

        EW_GREEN:
        begin
            NS = RED;
            EW = GREEN;
        end

        EW_YELLOW:
        begin
            NS = RED;
            EW = YELLOW;
        end

        default:
        begin
            NS = RED;
            EW = RED;
        end

        endcase

    end

end

endmodule
